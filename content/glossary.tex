\newcommand{\newglossarywacronym}[4][]{
  %%% The glossary entry the acronym links to
  \newglossaryentry{#2_gls}{
    name={#2},
    long={#3},
    description={#4}
  }

  % Acronym pointing to glossary
  \newglossaryentry{#2}{
    type=\acronymtype,
    name={#2},
    description={#3},
    first={\ifx\relax#1\relax#3\else#1\fi\ (#2)\glsadd{#2_gls}},
    see=[Glossary:]{#2_gls}
  }
}
